
\section{Différents arbres}


\subsection{Définitions}

\begin{frame}
  \frametitle{Pourquoi des arbres ?}

  \begin{itemize}
  \item<1-> Structure dynamique : liste
  \item<2-> Constat : coût élevé des opérations de base
  \item<3-> Objectif : trouver une structure de données plus efficace
  \end{itemize}

  \uncover<4->{Arbre~:}
  \begin{itemize}
  \item<5-> Structure de données dynamique
  \item<6-> Bonne complexité pour les opérations de base
  \item<7-> Recherche, insertion, suppression : $\Theta(\log n)$
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Définitions}

  \only<1,3,5,7,9,11>{
    \begin{defi}[Arbre]
      Ensemble non vide de noeuds et d'arêtes tel que~:
      \begin{itemize}[<+->]
      \item<3-> noeud : objet simple
      \item<5-> arête : relie deux noeuds (orienté)
      \item<7-> branche : suite de noeuds reliés par des arêtes
      \item<9-> racine : noeud sans prédécesseur
      \item<11-> tous les noeuds sont accessibles depuis la racine par une
        seule branche
      \end{itemize}
    \end{defi}}

\subsection{Arbres binaires}

\begin{frame}
  \frametitle{Définition}

  \only<1>{
    \begin{defi}[Arbre binaire]
      Un arbre binaire est un arbre dont chaque noeud possède au plus deux
      fils. 
    \end{defi}

    \begin{defi}[Arbre binaire complet]
      Un arbre est complet si chaque noeud qui n'est pas une feuille possède
      exactement deux fils et où toutes les feuilles sont au même niveau.
    \end{defi}
  }

\begin{frame}
  \frametitle{Propriétés}
  \begin{itemize}[<+->]
  \item Chaque noeud possède un fils gauche et un fils droit
  \item Arbre de hauteur $h$
  \item[$\Rightarrow$] $2^h-1$ noeuds (si complet)
  \item $n$ noeuds
  \item[$\Rightarrow$] hauteur $\log n$
  \end{itemize}
\end{frame}

  \begin{defi}[Arbre binaire de recherche]
    Arbre binaire vérifiant la propriété : pour tout noeud $n$ de l'arbre,
    tous les noeuds de son sous-arbre  gauche sont inférieurs à $n$,
    tous les noeuds de son sous-arbre droit sont supérieurs à $n$.
  \end{defi}

\subsection{Recherche}

\begin{frame}
  \frametitle{Recherche}
  Rechercher un élément~:
  \begin{itemize}[<+->]
  \item partir de la racine
  \item comparer la valeur recherchée à l'élément courant
    \begin{itemize}
    \item si elle est égale : fin
    \item si elle est inférieure : chercher dans le sous-arbre gauche
    \item sinon : chercher dans le sous-arbre droit
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Parcours exhaustif}

\begin{frame}
  \frametitle{Principe}
  \begin{itemize}[<+->]
  \item parcourir un arbre : appliquer un traitement à tous les noeuds
  \item algorithme itératif : difficile à écrire
  \item algorithme récursif : très simple
    \begin{itemize}
    \item principe~: 3 traitements $\Rightarrow$ 2 sous-arbres et noeud courant
    \item noeud courant avant le fils gauche $\Rightarrow$ préfixe
    \item noeud courant entre les fils  $\Rightarrow$ infixe
    \item noeud courant après le fils droit $\Rightarrow$ postfixe
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Arbres équilibrés}

\begin{frame}
  \frametitle{Intérêt}
  \begin{itemize}
  \item Arbre binaire : efficace si complet
  \item Arbre binaire complet : pas toujours possible
  \item[$\Rightarrow$] notion d'arbre équilibré
  \end{itemize}
  \pause
  \begin{defi}[Arbre équilibré]
    Arbre binaire est équilibré : la différence entre les
    hauteurs des fils gauche et droit de tout noeud ne peut excéder 1.
  \end{defi}

\end{frame}
