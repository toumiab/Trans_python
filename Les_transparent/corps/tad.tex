
\section{Type abstrait de données}


\subsection{Définition}

\begin{frame}
  \frametitle{Définition}

  \begin{defi}[Type abstrait de données]
    Concept d'utilisation de données ne tenant pas en compte leur
    réprésentation interne.
  \end{defi}

  \pause
  \begin{itemize}
  \item Conception d'un algorithme : démarche descendante \pause
  \item Initialement : on est loin de l'implantation\ldots{} \pause 
    puis on affine\pause 
  \item Étape initiale : type abstrait de données (TAD)\pause
  \item Définir les opérations sur le TAD
  \item Cacher la représentation interne\pause
  \item<alert@7> Objet simplifié 
  \end{itemize}
  \pause
  Utilité : briques réutilisables
\end{frame}

\begin{frame}
  \frametitle{Exemple : rationnels}
  \begin{itemize}[<+->]
  \item TAD nombre rationnel
  \item Opérations : 
    \begin{itemize}
    \item création
    \item extraction numérateur et dénominateur
    \item addition, soustraction, multiplication, division
    \item test d'égalité
    \end{itemize}
  \item Représentation interne : par exemple deux entiers
  \item Contrainte interne : numérateur strictement positif
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Exemple : rationnels}

  \begin{itemize}
  \item \only<1>{Définition d'un nouveau \textcolor{red}{type}}
\only<2>{Représentation \textcolor{red}{interne}}
\only<3-5>{\textcolor{red}{Initialisation} de la représentation interne}
\only<6>{Opérations}
  \item \only<1>{Classe}
\only<2>{Variables d'instance}
\only<3-5>{Constructeur}
\only<6>{Méthodes}
  \end{itemize}

  \begin{semiverbatim}
    {\color<1>{red}public class Rationnel} \{
\only<2>{

  \hspace{1cm}\textcolor{red}{private} int numerateur;

    \hspace{1cm}\textcolor{red}{private} int denominateur;}

    \hspace{1cm}...
\only<3>{

  \hspace{1cm}public \textcolor{red}{Rationnel}(int num, int den) \{

    \hspace{2cm}if (den == 0) \{

        \hspace{3cm}numerateur = 0;

        \hspace{3cm}denominateur = 0;

      \hspace{2cm}\}}
\only<4>{

    \hspace{2cm}else if(den > 0) \{

        \hspace{3cm}numerateur = num;

        \hspace{3cm}denominateur = den;

      \hspace{2cm}\}}
\only<5>{

    \hspace{2cm}else \{

        \hspace{3cm}numerateur = -num;

        \hspace{3cm}denominateur = -den;

      \hspace{2cm}\}

  \hspace{1cm}\}}
\only<6>{

  \hspace{1cm}public void \textcolor{red}{mult}(Rationnel r) \{

    \hspace{2cm}numerateur *= r.numerateur;

    \hspace{2cm}denominateur *= r.denominateur;

  \hspace{1cm}\}

  \}}
  \end{semiverbatim}
\vspace{5cm}
\end{frame}

\section{Notion d'objet}


\subsection{Concepts fondamentaux}

\begin{frame}
  \frametitle{Concepts}
  Objectif :
  \begin{itemize}
  \item Augmenter la réutilisabilité
  \item[$\Rightarrow$] modularité du code, notion de composants
  \end{itemize}

  \pause
  Concepts mis en place :
  \begin{itemize}
  \item {\color<3>{red}encapsulation}
  \item héritage
  \item polymorphisme
  \end{itemize}

  \uncover<4>{
    \begin{defi}[Encapsulation]
      Mécanisme permettent de cacher l'implantation d'un TAD. Fournit une
      interface pour accéder aux données.
    \end{defi}
  }

\end{frame}

\begin{frame}
  \frametitle{Notion de classe et d'instance}
  \begin{itemize}
  \item Classe $\simeq$ type
  \item Instancier la classe $\simeq$ déclarer une variable de ce type
  \item Objet : variable dont le type est une classe
  \end{itemize}
  \pause
  Classe $\leadsto$ plan

  Instance $\leadsto$ réalisation
  \pause
  \begin{alertblock}{Important}
    Un programme est composé d'instances et non de classes
  \end{alertblock}
\end{frame}

\begin{frame}
  \frametitle{Modificateurs d'accès}
  Quatre modificateurs d'accès possibles :
  \begin{itemize}[<+->]
  \item public : utilisable depuis toutes les classes
  \item private : utilisable uniquement dans sa classe
  \item protected : utilisable depuis la classe et les sous-classes
  \item default (rien) : utilisable dans le paquetage courant
  \end{itemize}
\end{frame}

  \end{itemize}

    \}
  \end{exemple}

\begin{frame}
  \frametitle{Contexte d'exécution}
  \begin{itemize}
  \item Une variable = une instance = un contexte d'exécution
  \item chaque variable contient ses données propres
  \end{itemize}

  \begin{exemple}
    \begin{semiverbatim}
      Rationnel x1 = new Rationnel(1,2);

      Rationnel x2 = new Rationnel(2,3);

      System.out.println(x2.getNumerateur());

      $\Rightarrow$ 1 2
    \end{semiverbatim}
  \end{exemple}
\end{frame}

\subsection{TAD Pile}

\begin{frame}
  \frametitle{Pile}
  
  \begin{itemize}
  \item Principe : empiler et dépiler des éléments par le haut de la pile
  \item Pile = LIFO (Last In First Out)
  \item Fonctionnalités :
    \begin{itemize}
    \item Empiler, dépiler
    \item Examiner le sommet (sans dépiler)
    \item Tester si la pile est pleine ou vide
    \end{itemize}
  \end{itemize}

\begin{frame}
  \frametitle{Représentation de la pile}
  Exemple : pile d'entiers
  \begin{itemize}
  \item Choix d'une représentation interne : tableau d'entiers + position du sommet de la pile
  \item Création d'une pile : allouer le tableau
  \item Empiler, dépiler, extraire le sommet : accéder au tableau
  \item Pile vide ou pleine : tester la taille du tableau
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Pile en Java}
  \begin{exemple}[Pile]
    \begin{semiverbatim}
public class Pile \{

\only<1>{  \hspace{1cm}private int contenu[];

  \hspace{1cm}private int sommet;}
\only<2>{  \hspace{1cm}public Pile(int taille)  \{

    \hspace{2cm}contenu = new int[taille];

    \hspace{2cm}sommet = -1; // pile vide

  \hspace{1cm}\}}
\only<3>{  \hspace{1cm}public boolean estVide()  \{

    \hspace{2cm}return sommet==-1;

  \hspace{1cm}\}

 \hspace{1cm} public boolean estPleine()  \{

    \hspace{2cm}return sommet==contenu.length-1;

  \hspace{1cm}\}}
\only<4>{  \hspace{1cm}public void empiler(int val) \{

    \hspace{2cm}if (!estPleine()) \{

        \hspace{3cm}sommet++; 

        \hspace{3cm}contenu[sommet] = val; 

      \hspace{2cm}\}

  \hspace{1cm}\}}
\only<5>{  \hspace{1cm}public int depiler()  \{

    \hspace{2cm}int val;

    \hspace{2cm}if (!estVide())  \{

        \hspace{3cm}val = contenu[sommet]; 

        \hspace{3cm}sommet--; 

      \hspace{2cm}\}

    \hspace{2cm}else

      \hspace{3cm}val = -1; 

    \hspace{2cm}return val;

  \hspace{1cm}\}}
\only<6>{  \hspace{1cm}public int getSommet()  \{

    \hspace{2cm}int val;

    \hspace{2cm}if (!estVide())

      \hspace{3cm}return contenu[sommet];

    \hspace{2cm}else

      \hspace{3cm}return -1; 

  \hspace{1cm}\}}

\}      
    \end{semiverbatim}
  \end{exemple}
\end{frame}

\subsection{TAD File}

\begin{frame}
  \frametitle{TAD File}
  \begin{itemize}
  \item Principe : Insérer et supprimer des éléments de la file
  \item File = FIFO (First In First Out)
  \item Fonctionnalités :
    \begin{itemize}
    \item Insérer, supprime
    \item Examiner la tête (sans défiler)
    \item Tester si la file est pleine ou vide
    \end{itemize}
  \end{itemize}

\begin{frame}

  Exemple : file d'entiers
  \begin{itemize}
  \item Représentation interne : liste chaînée, vecteur, tableau + 2 indices
  \item Création d'une file : initialiser la liste ou le vecteur, allouer le tableau
  \item Enfiler, défiler, extraire la tête : accéder au tableau
  \item File vide ou pleine : tester les indices d'entrée et de sortie
  \end{itemize}
