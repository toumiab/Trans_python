	\only<4-6>{
	\begin{itemize}
		\item <4-6> Sont représentés en en double précision (64 bits).\\
	$\Rightarrow$~{le plus petit nombre qu'il est possible de distinguer de 1 est $2.22E-16$}
	 \item <5,6> Pour utiliser la simple précision, on peut faire appel à la bibliothèque {\tt

		\end{itemize}
		}
	 \only<6>{    
			\begin{exemple}

		 }
	\item<7-> \textbf{Les complexes}
	\only<7-9>{
	\begin{itemize}
		\item <7-9>Sont formés d'un couple de réels qui composent la partie réelle et la partie imaginaire
		\item <8-9> Un suffixe \emph{j} pour regrouper deux valeurs composant la partie réelle et la partie imaginaire
		\item <9> Un nombre complexe possède deux attributs en lecture seule \emph{.real} et \emph{.imag} et une méthode \emph{.conjugate()}
	\end{itemize}
	}	 
		
		
		\only<10>{

	  }

\begin{frame}
\frametitle{Les booléens}

 \begin{itemize}[<+->]
	 \item Un objet booléen peut prendre 2 valeurs \emph{True} ou \emph{False}
	 \item Une variable booléenne supporte toutes les opérations logiques (and, or, not)
	 \item Tout objet peut être interprété en tant que valeur booléenne 
	\item Tout ce qui n'est pas faux est vrai
	\begin{itemize}
		\item l'objet None est faux
		\item toutes les valeurs numériques non nulles sont vraies (0, 0.0, 0.0+0.0j sont fausses)
		\item tous les agrégats (séquences, dictionnaires, ensembles) sont vrais s'ils contiennent au moins un élément, faux sinon
	\end{itemize}
 \end{itemize}


\end{frame}

\begin{frame}

\begin{remarques}
\begin{itemize}
	\item Les types simples (scalaires, atomiques) permettent de stocker une seule donnée par variable, contrairement aux types composites (containers) 
	\item Les types simples sont immutables (non modifiables)
	\end{itemize}
\end{remarques}
\end{frame}

\begin{frame}
\frametitle{Les séquences}
\begin{defi}[Une séquence]
    est une collection \emph{ordonnée} d'objets ou chaque objet est accessible via sont index (indice) 
  \end{defi}
	\pause
\begin{itemize}
	\item On distingue deux familles  de séquences
   
	 \begin{enumerate}
		 \item Les séquences immutables (non modifiables, non-mutables) : Les chaînes de caractères (\emph{str}) et les tuples (\emph{tuple})
		 \item Les séquences mutables (modifiables) : Les listes
	 \end{enumerate}
\end{itemize}
\pause
\begin{remarque}[]
    Toutes les opérations sur les séquences immutables sont disponibles sur les séquences mutables. 
  \end{remarque}

\end{frame}


\begin{frame}
\frametitle{Les séquences}
Indexation des séquences :
\begin{itemize}[<+->]
	\item Les séquences sont indexées en commençant par 0
	\item Chaque élément (objet) d'une séquence est accessible via son index
	\item Si l'index est négatif, on accède à la séquence à partir de la fin.\\
    \verb!\colorbox{pink}{>>> s[-1] \# \footnotesize{renvoie le dernier élément de la séquence s}}!
	 \item La tranche \texttt{[i:j]} désigne tous les éléments commençant de $i^e$ indice jusqu'au $(j-1)^e$ indice\\
     \verb!\colorbox{pink}{>>> s[0:2] \# \footnotesize{renvoie une séquence contenant s[0] et s[1]}}!
			
		  
			\begin{itemize}
				\item \verb![:j]! : désigne tout ce qui précède j ($j^e$ élément exclu)
				\item \verb![i:]! : désigne tout ce qui suit i ($i^e$ élément inclu)
				\item \verb![:]!  : tous les éléments (utilisé pour réaliser une copie \textit{artificielle})
				\item \verb![i:j:pas]! : pour réaliser une sélection avec \texttt{un pas} donné 
			\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
  \frametitle{N-Uplets (Les tuples)}
 
 \begin{defi}[Un tuple]
  Une séquence (collection) ordonnée \textbf{non modifiable} d'éléments hétérogènes (\textit{c'est une liste non modifiable}) 
	\end{defi}
	\pause
	\begin{itemize}
	\item Ils sont notés entre parenthèses \emph{()}, ou une suite d'éléments séparés par des virgules
	\item l'accès aux éléments du tuple est réalisé via les indices placés entre crochets\emph{ [ ] }(comme pour les listes).
	\end{itemize}
	\pause
\begin{remarques}
\begin{itemize}
	\item Si un seul élément est dans un tuple, le faire suivre d'une virgule.    
	\item Le tuple possède les mêmes \emph{méthodes} que la liste, à l'exception de celles permettant une modification.
\end{itemize}
\end{remarques}

\end{frame}

\begin{frame}
  \frametitle{Dictionnaire}
 \begin{defi}[Dictionnaire]

 Est une collection modifiable, \textbf{non ordonnée}  d'éléments hétérogènes ou chacun de ses éléments est indicé par une \emph{clé}.
 \end{defi}
\pause
\begin{itemize}
	\item Il est noté entre accolades \emph{\{\}}, ou crée à partir d'une liste de couples (clé, valeur) passée en argument à la fonction \emph{dict()}.\\ 
	\pause
	Exemple :\\
	\footnotesize{
	\colorbox{pink}{\parbox{9cm}{
	 \verb!dic = \{1: 'a', 2: 'b'\}\\
	 \textcolor[rgb]{1,0,0}{\# ou élément après élément :}\\
	 dic=\{\} ; dic[1] = 'a' ; dic[2] = 'b' \\
	 \textcolor[rgb]{1,0,0}{\# en utilisant la fonction dict()}\\
	 dic = \textcolor[rgb]{0,0,1}{dict}([(1, 'a'), (2, 'b')]) \\
   dic = \textcolor[rgb]{0,0,1}{dict}(\textcolor[rgb]{0,0,1}{zip}((1, 2), ('a', 'b')))\\
	 len(dic)  \textcolor[rgb]{1,0,0}{ \# => 2 paires clé:valeur} \\
	 1 \textcolor[rgb]{0,0,1}{in} dic   \textcolor[rgb]{1,0,0}{\# test existence clé => True }\\
   5 \textcolor[rgb]{0,0,1}{in} dic   \textcolor[rgb]{1,0,0}{\# => False, car 5 n'est pas une clé }
	 !}}
	}
\end{itemize}


\end{frame}



\begin{frame}
 
 \frametitle{Dictionnaire}
 \begin{propriete}[]
 \begin{itemize}
	\item Les \emph{clés} doivent être de type non modifiable (exemple : les types simples) et uniques. 
	\item Les valeurs peuvent être de n'importe quel type
	\item Les valeurs sont accessibles par leurs clés 
	\item Pas de notion d'ordre
	\end{itemize}
 
 \end{propriete}
\end{frame}

\begin{frame}
  \frametitle{Affectation}  
  \begin{defi}[Affectation (assignation)]
    Affectation : donner une ou plusieurs valeurs à une ou à plusieurs variables réalisée par l'opérateur``=''. Elle peut être simple ou multiple.
  \end{defi}
  \pause
  \begin{exemple}[Affectations]
    \begin{semiverbatim}
      \textcolor[rgb]{1,0,0}{\# affectation simple }
			
			a = 2 
      
			\textcolor[rgb]{1,0,0}{\# affectation multiple }
      
			a = b = 5 \textcolor[rgb]{1,0,0}{\# => a=5 et b=5}
			
			a, b = b, a \textcolor[rgb]{1,0,0}{\# échange le contenu de a et b}

      c, d, e = 1, 5, 6 \textcolor[rgb]{1,0,0}{\# => c=1; e=5 et e=6}
			
			a, *reste = [1, 2, 3] \textcolor[rgb]{1,0,0}{\# => a=1 et reste=[2, 3]}
			

    \end{semiverbatim}
  \end{exemple}
\end{frame}

  \item Il existe deux types de boucles en Python 3 : for, while
	\item Les deux boucles sont équivalentes
  \end{itemize}

\begin{frame}
  \frametitle{while}
  \begin{semiverbatim}
    \textcolor[rgb]{0,0,1}{while} condition :

      \hspace{1cm}actions

    
  \end{semiverbatim}
  \pause
  \begin{exemple}[boucle while]
    \begin{semiverbatim}
		
		i=0
		
		\textcolor[rgb]{0,0,1}{while} i<10:  \textcolor[rgb]{1,0,0}{\# boucle tant que i<10}
		
    \hspace{1cm}print(i)             \textcolor[rgb]{1,0,0}{\# affiche la valeur de i}
		
    \hspace{1cm}i = i+1              \textcolor[rgb]{1,0,0}{\# incrémente i}
		
    \textcolor[rgb]{1,0,0}{\# maintenant i vaut 10}

    \end{semiverbatim}
  \end{exemple}
\end{frame}

		\item Le mot clé \emph{continue} permet de passer immédiatement à l'itération suivante
	
	\end{itemize}

		\hspace{2cm}break
		
    \end{semiverbatim}
  \end{exemple}
\end{frame}


\begin{frame}

  Syntaxe~:
  \begin{semiverbatim}
    \textcolor[rgb]{0,0,1}{for} variable in collection:
		
		 \hspace{1cm}actions \textcolor[rgb]{1,0,0}{\# corps de la boucle}
   \end{semiverbatim}  
	\only<1-5>{
   \begin{exemple} 
	  \begin{semiverbatim}
		
		\only<1>{s = ['septembre', 'octobre']}
		\only<2>{s = "Python"}
		\only<3>{s = 1,2,3,4}
		\only<4-5>{s = \{'a':10, 'b':'deux'\}  }
		
		\only<5>{for \textcolor[rgb]{0,0,1}{n, val} in s.items():}\only<1-4>{\textcolor[rgb]{0,0,1}{for} n \textcolor[rgb]{0,0,1}{in} s:} \only<4>{\textcolor[rgb]{1,0,0}{\# s.values(), s.keys(), s.items()}}
    		
		\hspace{1cm} print(n, end=' ')   \only<1>{\textcolor[rgb]{1,0,0}{\# => septembre octobre}}\only<2>{\textcolor[rgb]{1,0,0}{\# => P y t h o n}} \only<3>{\textcolor[rgb]{1,0,0}{\# => 1 2 3 4 }}\only<4-5>{\textcolor[rgb]{1,0,0}{\# => a b }}
   
	\end{semiverbatim}  
	  
	 
	\end{exemple}
	}
		\only<6>{
	\begin{remarque}[for : sortie de boucle]
 Possibilité de modifier l'exécution d'une boucle \emph{for} avec \emph{break} et
  \emph{continue} comme dans le cas de la boucle while.
 \end{remarque}
	}

	\begin{remarque}
	
	\begin{itemize}
		\item L'itérateur \emph{range(deb,fin, inc) }permet d'itérer sur une suite de nombres entiers, 
		\item La fonction \emph{enumerate(sequence)} retourne un objet permettant d'itérer sur l'indice et la valeur d'une séquence
	\end{itemize}  
	\end{remarque}
	
  	\begin{exemple}
		\begin{semiverbatim}
      s = ['septembre', 'octobre'] 
		 
		 \only<1>{for i in \textcolor[rgb]{0,0,1}{range(2)}:}\only<2>{for ind, val in \textcolor[rgb]{0,0,1}{enumerate(s)}:}
		
		 	\only<1>{\hspace{1cm} print(s[i], end=' ')   \textcolor[rgb]{1,0,0}{\# => septembre octobre}}
		 \only<2>{\hspace{1cm}print(ind, val,  end = ' ')}
		 
		 \only<2>{\hspace{1cm}\textcolor[rgb]{1,0,0}{\# => 0 septembre 1 octobre }}
			
    \end{semiverbatim}

	\end{exemple}

\end{frame}


\begin{frame}
  \frametitle{for : forme condensée}
\textbf{ Objectif : }Pour construire un container de type liste, dictionnaire ou set  à l'aide d'une boucle for.\\
\pause
Syntaxe :

\texttt{liste = [expression \emph{for} expr \emph{in} iterable \emph{if} cond]}\\
\texttt{dict = \{expr1:expr2 \emph{for} expr \emph{in} iterable \emph{if} cond\}}

\pause 
Exemple 1 :\\
\colorbox{pink}{\parbox{10.5cm}{
	     \verb!puiss2 = [nb*nb \emph{for} nb \emph{in} \emph{range}(1,11) \emph{if} nb\%2==0]\\
			 \textcolor[rgb]{1,0,0}{\# => [4, 16, 36, 64, 100]}!}}\\
			\pause

Exemple 2 :\\
\colorbox{pink}{\parbox{10.5cm}{
	     \verb!dic = \{'a': 12, 'b': 1.5, 'c': 3\}\\
        dic1 = \{cle:val \emph{for} cle,val \emph{in} dic.items() \emph{if} val**2>10\}
				\textcolor[rgb]{1,0,0}{\# => dic1 = \pause \{'a': 12\}}!}}





\end{frame}
