\begin{itemize}
\item<3-> \textbf{Les types intégrés de Python}
\begin{itemize}
    \item<4-> Entiers signés (complément à deux) : le type \emph{int} 
    \item<4-> Réels (IEEE 754) : le type \emph{float} en double précision (64 bits)
    \item<4-> Complexes : le type \emph{complex}
		\item<5-> Booléens : le type \emph{bool}
		\item<6-> Chaînes de caractères : type \emph{str}
		\item<7-> Listes : type \emph{list}
		\item<8-> Dictionnaires : type \emph{dict}
		\item<9-> Tuples : type \emph{tuple}
		\item<10-> Ensembles modifiables et immuables : le type \emph{set} et le type \emph{frozenset}
    \end{itemize}
\end{itemize}
}
