\begin{frame}

  \begin{defi}[Liste chaînée]
    Liste chaînée : structure algorithmique \emph{dynamique}. Accès
    \emph{séquentiel} aux éléments.
  \end{defi}

    \begin{itemize}

\begin{frame}
  \frametitle{TAD liste}
  \begin{itemize}[<+->]
  \item Liste composée d'éléments (noeuds)
  \item Noeuds chaînés entre eux :
  \item Chaque noeud connaît son successeur
  \item Fin de liste : \emph{None}
  \item Chaque noeud peut connaître son prédécesseur (liste doublement
    chaînée)
  \item Liste = premier élément (noeud) + opérations
    \begin{itemize}
    \item Insertion
    \item Suppression
    \item Recherche
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Classe noeud}
  \begin{itemize}
  \item Constructeur (Noeud) : 
	\item $\Longrightarrow$Noeud : données\uncover<2->{ + variable pour le chaînage}
	\item <3>Les opérations sur les noeuds : affichage, comparaison (==, $>$, $>$,$<=$, ...) 
  \end{itemize}

  \begin{exemple}
    \begin{semiverbatim}
      \vspace{-0.5cm}
      class Node (object):
			
      \hspace{.5cm}def \_\_init\_\_(self, donnee = 0):
			
			 \hspace{1cm}\textcolor[rgb]{0,0.58,0}{""" Creation d'un noeud}
			
		   \hspace{1cm}\textcolor[rgb]{0,0.58,0}{le noeud suivant est fixé à None}
			
			 \hspace{1cm}\textcolor[rgb]{0,0.58,0}{"""}
			
      \hspace{1cm}self.\textcolor<1>{red}{val} = donnee \textcolor[rgb]{0,0.58,0}{\# La valeur de noeud}
			
			\uncover<2->{
      \hspace{1cm}{self.\color<2>{red}next} = None  \textcolor[rgb]{0,0.58,0}{\# Le noeud suivant}}
      \uncover<3->{
			
			\hspace{.5cm} \textcolor[rgb]{1,0,0}{...}}
			
    \end{semiverbatim}
  \end{exemple}
\end{frame}

\begin{frame}

  \only<1>{
\begin{algorithm2e}[H]
  \caption{Parcours de liste}
  \KwIn{premier : noeud \tcp{ tête de liste}} 
  \KwData{n : noeud \tcp{ noeud courant}}
  $n \gets premier$ \;
  \While{$n \neq None$} {
    $n \gets n.suivant $\;
  }

\begin{frame}
  \frametitle{Recherche d'un élément dans une liste}
  Exemple de parcours~: recherche d'élément\\
  Principe~:
  \begin{itemize}[<+->]
  \item Partir du début de la liste
  \item Avancer tant que l'élément n'est pas trouvé
  \item Et que la fin de liste n'est pas atteinte
  \item Si l'algorithme se termine sur la fin de liste, l'élément n'est pas
    présent
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Algorithme}
\begin{algorithm2e}[H]
  \caption{Recherche d'élément dans une liste}
    \KwIn{int val}
    \KwData{debut : noeud \tcp{ tête de liste}}
    \KwData{n : noeud \tcp{ noeud courant}}
    $n \gets debut$ \;
    \While{$n \neq None$ et $n.valeur \neq val$} {
      $n \gets n.suivant$ \;
    }
    retourner \emph{n} \;
\end{algorithm2e}

\begin{frame}
  \frametitle{Remarque sur l'insertion dans une liste triée}
  \begin{remarque}
    \begin{itemize}[<+->]
    \item Deux cas particuliers à tester
    \item Ruse :
      \begin{itemize}
      \item créer un élément temporaire \emph{tmp}
      \item accrocher le début de la liste après \emph{tmp}
      \item parcourir la liste à partir de \emph{tmp}
      \end{itemize}
    \item[$\Rightarrow$] plus de cas particulier !
    \end{itemize}
  \end{remarque}
\end{frame}

  $n \gets tmp$\;\pause
  \While{$n.suivant \neq None$ et $n.suivant < nouv$} {
    $n \gets n.suivant$ \;
  }\pause
  $nouv.suivant \gets n.suivant$\;
  $n.suivant \gets nouv$\;
  $debut \gets tmp.suivant$ \tcp{ supprimer l'élément temporaire}
\end{algorithm2e}

\subsection{Suppression}


\begin{frame}
  \frametitle{Suppression}
  Principe~:
  \begin{itemize}
  \item Partir du début de la liste
  \item Se placer avant l'élément à supprimer
  \item Modifier le chaînage pour supprimer l'élément
  \end{itemize}

    }

    }

  }
  \only<5->{
    \begin{remarques}

\subsection{Listes circulaires}

\begin{frame}
  \frametitle{Principe}
  \begin{itemize}
  \item Le successeur d'un noeud peut être un noeud précédent
  \item Attention lors des parcours
  \item[$\Rightarrow$] risque de boucle infinie
  \end{itemize}

\begin{frame}
  \frametitle{Détection de cycle}
  \begin{itemize}[<+->]
  \item Problème : comment détecter que la liste contient un cycle ?
  \item Solution : 
    \begin{itemize}
    \item utiliser deux variables :
    \item une qui avance d'un noeud par tour
    \item une qui avance de deux noeuds par tour
    \item si un aboutit à la fin de la liste : liste non circulaire
    \item si les deux se rejoignent : liste circulaire
    \end{itemize}
  \end{itemize}
\end{frame}
