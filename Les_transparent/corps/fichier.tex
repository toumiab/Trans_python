\subsection{Traitement des exceptions}

\begin{frame}
  \frametitle{Traitement}%\chrono}
 
  Deux possibilités :
  \begin{itemize}
  \item<1-| alert@3> traiter l'exception localement ;
  \item<2-| alert@4> transmettre l'exception à la fonction appelante.
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Traitement local : \emph{try-except}}%\chrono}

  Traiter une exception localement : utiliser le bloc \emph{try-except-finally}.
  \begin{itemize}
  \item try : certaines instructions peuvent lever une exception
  \item except : traitement d'une exception
	\item else (optionnelle). Instructions exécutées si aucune exception n'est levée
  \item finally (optionnelle): instruction exécutées après le bloc \emph{try} et les
    \emph{except} éventuels.
	
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Syntaxe}%\chrono}
  Syntaxe :
  \begin{semiverbatim}
	
\textcolor[rgb]{0,0,1}{try} :\textcolor[rgb]{0,0.58,0}{
    \# Cette fonction peut lever une exception}
    fonction_levant_une_exception()
\textcolor[rgb]{0,0.58,0}{\# Ici on récupère toutes les exceptions}
\textcolor[rgb]{0,0.58,0}{\# héritant de Exception}
\textcolor[rgb]{0,0,0.55}{except} Exception as err:
    \textcolor[rgb]{0,0.58,0}{\# Traitement de lexception}
    print (err)
\textcolor[rgb]{0,0.58,0}{\# Suite du programme}
  \end{semiverbatim}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Syntaxe}%\chrono}

  \begin{semiverbatim}

    # ...

    # ...

    # ...

    # ...
  \end{semiverbatim}
\end{frame}

\subsection{Fichiers binaires}

\begin{frame}[fragile]
  \frametitle{Lecture de fichiers binaires}%\chrono}
  \begin{itemize}
  \item Un fichier est lu octet par octet 
	\item Nécessite la connaissance de sa structure interne
	\item La méthode \emph{.read(nbOcetets)} pour lire des octets.
  \end{itemize}
	\only<2>{
\begin{remarque}
\begin{itemize}
	\item Utilisation de la méthode \emph{.from\_bytes()} qui convertit des octets en variable entière.
	\item Formatage des octets lus : struct.unpack() \\
	 Exemple : \\
	\texttt{a, b, c = struct.unpack("<BBBx",file.read(0x4))))}
\end{itemize}
\end{remarque}
}
	
\begin{exemple}<3>
\begin{semiverbatim}\scriptsize{
\vspace{-.5cm}
# Ouverture du fichier binaire en lecture
\emph{with} open(filename, mode ='\textcolor[rgb]{1,0.41,0.13}{rb}') as f: 
    \textcolor[rgb]{0.24,0.7,0.44}{# Lecture du nombre de données}
    octets = f.\textcolor[rgb]{1,0.41,0.13}{read(4)}
    \textcolor[rgb]{0.24,0.7,0.44}{# Conversion du nombre de données en entier}
    nb = int.\textcolor[rgb]{1,0.41,0.13}{from_bytes}(octets, 'big')
    \textcolor[rgb]{0.24,0.7,0.44}{# nb = struct.unpack("<I",f.read(0x4))))}
    \textcolor[rgb]{0,0,1}{for} \_ in range (nb):
        octets = f.\textcolor[rgb]{1,0.41,0.13}{read}(4)
        val = int.\textcolor[rgb]{1,0.41,0.13}{from\_bytes}(octets, 'big')
        print(val)}

\end{itemize}
\end{remarque}

\subsection{Sérialisation}


\begin{frame}
  \frametitle{Principe}%\chrono}
  \begin{itemize}
  \item Sauvegarder directement des objets dans un fichier
  \item Puis les récupérer très simplement
  \item Utilisation du module \emph{pickel}
  \item Lecture/écriture du fichier par
        \emph{.load()}/\emph{.dump()}.
  \end{itemize}
	\pause
\begin{alertblock}{Attention}
	Ce qui a été sauvé avec pickle doit être chargé avec pickle. 
\end{alertblock}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Utilisation du module \texttt{pickle}}%}%\chrono}
  \begin{enumerate}
  \item<2-> Sauvegarder les objets
  \item<4-> Restaurer les objets
  \end{enumerate}

	\begin{exemple}
  \begin{semiverbatim}
	\vspace{-1cm}
\textcolor[rgb]{1,0.41,0.13}{import pickle}
\textcolor[rgb]{0,0.58,0}{# exemple d'un objet
}departement = \{36:'Indre',30:'Gard',75:'Paris'\}
\textcolor[rgb]{0,0.58,0}{# ouverture de fichier en écriture binaire}
with open('data.bin','wb') as f:
    \only<2->{\textcolor[rgb]{1,0.41,0.13}{pickle.dump}(departement,f) }
\only<3->{\textcolor[rgb]{0,0.58,0}{# ouverture de fichier en lecture binaire}
\textcolor[rgb]{0,0,1}{with} open('data.bin','\textcolor[rgb]{1,0.41,0.13}{rb}') as f:}
    \only<4->{dep = \textcolor[rgb]{1,0.41,0.13}{pickle.load}(f) }
  \end{semiverbatim}
	\end{exemple}

\end{frame}



\begin{frame}
  \frametitle{Utilisation du module \texttt{pickle}}%\chrono}
  
\begin{alertblock}{Remarques}
\begin{itemize}
	\item La méthode .\emph{dump}(variable,fichier) permet de stocker la variable.
	\item Pour stocker plusieurs variables, il suffit d'appeler plusieurs   fois la méthode \emph{.dump}
	\item Pour lire plusieurs variables stockées, il suffit d'appeler plusieurs fois la méthode \emph{.load}
	
\end{itemize}
\end{alertblock}
	
\end{frame}
