\subsection{Strategie}

\begin{frame}
  \frametitle{Objectif}
  \begin{itemize}
  \item Un objet possède une stratégie (comportement)
  \item Stratégie peut changer dynamiquement
  \end{itemize}
  Utilité~:
  \begin{itemize}
  \item Plusieurs classes ne diffèrent que par leur comportement
  \item Plusieurs variantes du comportement
  \end{itemize}
  Exemple~:
  \begin{itemize}
  \item Besoin de trier des vecteurs
  \item Plusieurs normes pour mesurer les vecteurs
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Principe}
  \begin{itemize}
  \item Classe possédant une stratégie
  \item Interface stratégie $\leadsto$ méthodes abstraites
  \item L'utilisateur écrit les classes utilisant la stratégie
  \item[$\Rightarrow$] obligation de redéfinir les méthodes
  \item Changement de stratégie~: instancier une nouvelle classe
  \end{itemize}
\end{frame}

    SortStrategy[] alg = {new ShellSort(), 
      new QuickSort() };
    // parcours de l'ensemble des tris
    for (SortStrategy str : alg) {
      // initialiser le ArraySort

\subsection{État}

\begin{frame}
  \frametitle{Objectif}
  \begin{itemize}
  \item Associer un comportement à un objet
  \item Comportement = états + transitions
  \item Dans chaque état~: une action
  \item[$\leadsto$] Automate
  \end{itemize}
  Codage~:
  \begin{itemize}
  \item Classe abstraite État
  \item + sous-classes
  \end{itemize}
\end{frame}

public class Normal extends AbstractState {
  public void action(Animal a) {
    if (a.detectPredator()) {
      // nouveau comportement : fuite
      a.setMyBehavior(new Scared());
    } else if (a.isHungry()) {
      // nouveau comportement : faim
      a.setMyBehavior(new Hungry());      
    }
  }
}

\subsection{Singleton}

\begin{frame}
  \frametitle{Singleton}
  \begin{block}{Objectif}
    Garantir qu'une classe ne pourra être instanciée qu'une seule fois.
  \end{block}
  Réalisation~?
  \begin{itemize}
  \item Classe \emph{Singleton}
  \item Variable de classe de type \emph{Singleton}
  \item Constructeur \emph{privé}
  \item Méthode \emph{de classe} d'instanciation
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Singleton en Java}
  \begin{lstlisting}
public class Singleton {
    /** singleton cree */
    private static Singleton s = null;
    /** constructeur prive */
    private Singleton ()  {
    }
    /** methode de classe de creation */
    public static Singleton getSingleton () {
      // creer le singleton si necessaire
      if (s == null) 
        s = new Singleton();
      return s;
    }
  }
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Utilisation du singleton en Java}
  \begin{lstlisting}
    public static void main (String[] args)  {
      Singleton s = Singleton.getSingleton(); 
      System.out.println(s);
      Singleton r = s.getSingleton(); // r = s
      System.out.println(r);
    }    
  \end{lstlisting}
\end{frame}

\subsection{Observer--Observable}

\begin{frame}
  \frametitle{Objectif}
  \begin{itemize}
  \item Simplifier la conception d'IHM
  \item Mécanisme de mise à jour d'interfaces observant des données
  \item Ajout de plusieurs vues très simple
  \end{itemize}
  Principe~:
  \begin{itemize}
  \item Données~:
    \begin{itemize}
    \item hériter de \emph{Observable}
    \item signaler des mises à jour avec \emph{setChanged} et \emph{notifyObservers}
    \end{itemize}
  \item Vues~:
    \begin{itemize}
    \item implanter \emph{Observer}
    \item réaliser les mises à jour dans \emph{update}
    \end{itemize}
  \end{itemize}
\end{frame}
