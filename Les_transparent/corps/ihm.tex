\subsection{Généralités}


\begin{frame}
  \frametitle{Pourquoi une IHM ?}
  Intérêt : 
  \begin{itemize}[<+->]
  \item Présenter des informations de manière synthétique
  \item Permettre d'interagir avec le programme
  \end{itemize}

  \begin{itemize}[<+->]
  \item Code sans difficulté technique
  \item Code souvent long et peu lisible

  \item Consulter la documentation de l'API
  \end{itemize}

\subsection{Classe interface graphique}

\begin{frame}[fragile]
  \frametitle{Définition d'une classe}

  \begin{itemize}
  \item Classe héritant de \emph{JFrame}
  \item Variables d'instance~: divers composants de l'interface
  \item Constructeur~: construction de l'interface
  \end{itemize}

  Exemple~:
  \begin{lstlisting}[frame=single]
public class Visu extends JFrame {
  private JButton boutonOK;
  public Visu() {
    boutonOK = new JButton();
    boutonOK.setText("OK");
    this.getContentPane().add(
          boutonOK,BorderLayout.SOUTH);
  }
  \end{lstlisting}
\end{frame}

\section{Gestion des évènements}


\subsection{Évènements associés à des objets}

\begin{frame}
  \frametitle{Principe}
  Principe général~:
  \begin{itemize}
  \item Ajouter un ``écouteur'' à l'objet
  \item Écouteur : classe implantant un \emph{Listener} ou héritant d'un
    \emph{Adapter}
  \item  interface \emph{ActionListener}~: redéfinir
    \emph{actionPerformed}
  \item interface \emph{MouseListener} ou classe \emph{MouseAdapter}~:
    redéfinir \emph{mouseClicked, mouseEntered, mouseExited, mousePressed} et
    \emph{mouseReleased}
  \item interface \emph{KeyListener} ou classe \emph{KeyAdapter}~: redéfinir
    \emph{keyPressed}, \emph{keyReleased} et \emph{keyTyped}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Exemple}
  Écouteur~:
  \begin{lstlisting}[frame=single]
class unEcouteur implements ActionListener {
  public void actionPerformed(ActionEvent e) {
    System.out.println("clic");
  }
}
  \end{lstlisting}

  Utilisation~:
  \begin{lstlisting}[frame=single]
JButton monBouton = new JButton();
monBouton.addActionListener(new unEcouteur());
  \end{lstlisting}
\end{frame}


\begin{frame}
  \frametitle{Critique de la méthode}
  Avantage~:
  \begin{itemize}
  \item Principe simple
  \end{itemize}
  \pause
  Inconvénient~:
  \begin{itemize}
  \item Obligation d'écrire une classe par écouteur !
  \item Problème de partage des données
  \end{itemize}
  \pause
  Solution~:
  \begin{itemize}
  \item Utiliser des classes internes
  \end{itemize}
\end{frame}


\subsection{Classes internes}


\begin{frame}
  \frametitle{Principe des classes internes}
  \begin{itemize}
  \item Principe : définir une classe à l'intérieur d'une autre classe
  \item[$\Rightarrow$] partage des données
  \item La classe interne peut utiliser les attributs privés de la classe
    externe.
  \item La classe interne utilise les données de la classe externe comme les
    siennes.
  \item Un objet interne est lié à un objet externe.
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Exemple de classe interne}
  \begin{lstlisting}[frame=single]
public class MaClasseExterne {
  private int x;
  private MaClasseInterne y = 
             new MaClasseInterne();
  public void action() {
    y.go();
  }

  class MaClasseInterne {
    void go() {
      x=42;
    }
  }
}
  \end{lstlisting}
\end{frame}

\begin{frame}
  \frametitle{Gestion d'évènements grâce à des classes internes}
  Application des classes internes~:
  \begin{itemize}
  \item gestion des évènements
  \item une classe interne par écouteur
  \end{itemize}
  Avantage~:
  \begin{itemize}
  \item plus de problème de partage de données
  \end{itemize}
  Inconvénient~:
  \begin{itemize}
  \item toujours une classe par écouteur
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Gestion d'évènement et classe interne}
  \begin{lstlisting}[frame=single]
public class Visu extends JFrame {
  private JLabel monTexte;
  private JButton boutonGo;
  public Visu() {
    ...
    boutonGo.addActionListener(new EcouteurGo());
  }

  class EcouteurGo implements ActionListener {
    public void actionPerformed(ActionEvent e) {
      monTexte.setText("Clic");
    }
  }
}
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Autre possibilité}
  \begin{itemize}
  \item possibilité : créer des objets dans créer de classe
  \item même principe qu'avec les classes internes
  \item possibilité d'utiliser les variables de la classe externe
  \end{itemize}

  \begin{lstlisting}[frame=single]
boutonFin.addActionListener(new ActionListener(){
  public void actionPerformed(ActionEvent e) {
    System.exit(0);
  }
});
  \end{lstlisting}
\end{frame}


\begin{frame}
  \frametitle{Évènements liés à la fenêtre}
  Il existe des évènements liés à la fermeture d'une fenêtre~:
  \begin{itemize}[<+-|alert@+>]
  \item quitter l'application : \emph{EXIT\_ON\_CLOSE}
  \item ne rien faire : \emph{DO\_NOTHING\_ON\_CLOSE}
  \item cacher la fenêtre : \emph{HIDE\_ON\_CLOSE}
  \item cacher et libérer la fenêtre : \emph{DISPOSE\_ON\_CLOSE}
  \end{itemize}

  \begin{semiverbatim}
    this.setDefaultCloseOperation(

    \hspace{2cm}\only<1>{JFrame.EXIT\_ON\_CLOSE}\only<2>{WindowConstants.DO\_NOTHING\_ON\_CLOSE}\only<3>{WindowConstants.HIDE\_ON\_CLOSE}\only<4>{WindowConstants.DISPOSE\_ON\_CLOSE});
  \end{semiverbatim}
\end{frame}

\begin{frame}

    \hspace{0.8cm}...

  \end{semiverbatim}
\end{frame}


\section{Dessin dans une fenêtre}


\subsection{Utilisation de la classe Graphics}

\begin{frame}
\frametitle{Généralités}
  \begin{itemize}
  \item API \emph{awt}~: méthodes de dessin en 2D
  \item Zone de dessin~: conteneur. Par ex. \emph{JFrame} ou \emph{JPanel}
  \item Objet permettant de dessiner~: \emph{Graphics2D}
  \item Possibilité de définir une classe (interne ou non) héritant de
    \emph{JPanel} 
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Graphics}
  Quelques méthodes de la classe \emph{Graphics}~:
  \begin{description}[<+->][drawRect, drawOval, drawLine]
  \item[clearRect] : efface une partie de l'image
  \item[drawRect, drawOval, drawLine] : affiche une figure
  \item[drawImage] : dessine une image
  \item[drawString] : affiche un texte
  \item[fillRect, fillOval, fillArc] : dessine et colorie une figure
  \item[setColor] : modifie la couleur du pinceau
  \end{description}\pause
  Voir également \emph{Graphics2D}.
\end{frame}


\subsection{Rafraichissement des fenêtres}

\begin{frame}
  \frametitle{repaint}

  Rafraîchissement d'une fenêtre graphique~:
  \begin{itemize}
  \item Appeler la méthode \emph{paint(Graphics g)}
  \item Cette méthode appelle \emph{paintComponent(Graphics g)}
  \end{itemize}

  Pour chaque composant (particulièrement les conteneurs)
  \begin{itemize}
  \item Possibilité de redéfinir \emph{paint/paintComponent}
  \end{itemize}
\pause
  \begin{remarque}
    Ne jamais appeler explicitement \emph{paint/paintComponent}. Utiliser
    \emph{repaint}.
  \end{remarque}
\end{frame}

\begin{frame}
  \frametitle{Exemple d'animation}
  \begin{semiverbatim}

\section{Conception de programmes avec interface graphique}

\begin{frame}
  \begin{alertblock}{Important}
    Séparer au maximum le programme de l'interface graphique. \pause
    Séparer ce que fait le programme de la manière dont il le représente.
  \end{alertblock}\pause

  \begin{itemize}[<+->]
  \item Exemple~: projets d'informatique.
  \item[$\Rightarrow$] Question : comment
    lier le programme et l'interface \emph{a posteriori} ?
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Lien programme / interface graphique}

  Programme : \emph{Simulation}
  \begin{itemize}[<+-|alert@+>]
  \item Interface : variable de type \emph{Simulation}
  \item Méthodes des boutons : appeler des méthodes de \emph{Simulation}
  \end{itemize}

  \begin{semiverbatim}
    public class Visualisation extends JFrame \{

    \hspace{1cm}...

\only<1>{    \hspace{1cm}private Simulation leProgramme;}
\only<2>{\hspace{1cm}public void actionPerformed(ActionEvent e) \{

          \hspace{2cm}leProgramme.simule();

          \hspace{2cm}repaint();

        \hspace{1cm}\}}
  \end{semiverbatim}
\end{frame}


\begin{frame}
  \frametitle{Remarques sur la création d'IHM}
  \begin{itemize}[<+->]
  \item Travail fastidieux
  \item Beaucoup de traitements systématiques
  \item Éditeur \emph{WYSIWYG} très utile
  \item[$\Rightarrow$] solution : utiliser un IDE
  \item Exemples~:
    \begin{itemize}
    \item \emph{eclipse} avec le \emph{plugin jigloo} (\url{http://www.cloudgarden.com/jigloo/})
    \item \emph{eclipse} avec le \emph{plugin VE}
%    \item \emph{jbuilder}
    \item \emph{NetBeans} (\url{http://netbeans.org/})
    \end{itemize}
  \end{itemize}

  
\end{frame}
