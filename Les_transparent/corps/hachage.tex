\begin{frame}
  \frametitle{Algorithmes}
  Accéder à la table de hachage~:
  \begin{itemize}
  \item Calculer le code $k$ associé à l'élément
  \item Accéder à la liste de la case $k$ du tableau
  \item cf. algorithmes sur les listes chaînées
  \end{itemize}
  Calcul de la clé~:
  \begin{itemize}
  \item Utilisation d'un tableau de $n$ cases
  \item Fonction de hachage~:\[
\begin{array}{llll}
  h: & {\mathcal K} & \rightarrow & [0,n-1] \subset \nbN\\
     & e & \mapsto & h(e)
\end{array}
    \]
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Hachage}
  \only<1,3>{
    Taux de remplissage~:
    \begin{itemize}
    \item Table $T$, $m$ alvéoles, $n$ éléments
    \item Taux de remplissage~: $\alpha=n/m$
    \end{itemize}
    \begin{itemize}
    \item $h(x)$ détermine une case du tableau
    \item[$\Rightarrow$] $\forall x, h(x) < n$
    \item Cas extrême~: uniquement des collisions $\leadsto \Theta(n)$
    \item<3> Cas extrême~: répartition uniforme  $\leadsto \Theta(1+\alpha)$
      % \item Comment obtenir cette fonction~?
      % \item[$\leadsto$] utilisation du modulo (\texttt{\%})
      % \item Il est conseillé d'adapter la fonction à chaque problème
    \end{itemize}
  }

\begin{frame}
  \frametitle{Définition d'une fonction de hachage}
  \only<1>{
  \begin{itemize}
  \item Bonne fonction de hachage~?
  \item[$\leadsto$] bonne répartition ($\simeq$ répartition uniforme)
  \end{itemize}
    \begin{exemple}[Méthode de la division]
      \begin{itemize}
      \item Associer un entier $k$ à un élément
      \item $h(k) = k \mod m$
      \item Attention au choix de $m$
        \begin{itemize}
        \item Éviter $m=2^p$, éviter les nombres proches des puissances de 2
        \item Bon choix~: nombre premier éloigné des puissances de 2
        \item Exemple~: 2000 éléments, $\alpha\simeq 3$ accepté $\leadsto m=701$
        \end{itemize}
      \end{itemize}
    \end{exemple}
  }
  \only<2>{
    \begin{exemple}[Méthode de la multiplication]
      \begin{itemize}
      \item Soit $A \in ]0,1[$, $h(k) = \lfloor m (kA \mod 1) \rfloor$
      \item Choix de $m$ non critique ; en général $m=2^p$
      \item Choix de $A$~?
        \begin{itemize}
        \item On veut $k$ sur un mot de $w$ bits
        \item $A = s/2^w$ avec $0<s<2^w$
        \item Knuth suggère $A\approx(\sqrt{5}-1)/2$
        \end{itemize}
      \item Ex.
        \begin{itemize}
        \item $k=123456, p=14, m=2^{14}=16384, w=32$
        \item $A=2654435769/2^{32} \approx (\sqrt{5}-1)/2$
        \item $m(kA \mod 1)=67.187744140625 \leadsto h(k)=67$
        \end{itemize}
      \end{itemize}
    \end{exemple}
  }
\end{frame}

\begin{frame}
  \frametitle{Exemple}
  \only<1>{
    \begin{exemple}[Hachage de mots]
      \begin{itemize}
      \item Mot~: suite de lettres ($w=a_0a_1a_2\ldots a_{p-1}$)
      \item Lettres codées par des entiers (ex. \textit{ASCII})
      \item Si uniquement lettres minuscules, utiliser $[0,25]$
      \item On note $'a'$ le code \textit{ASCII} de la lettre a
      \item Exemple~: coder "secret", "cretes"
      \end{itemize}
    \end{exemple}
  }
  \only<2>{
    \begin{alertblock}{Mauvaise fonction~: $h(w) = \sum_{i=0}^{p-1}a_i \mod m$}
      \begin{itemize}
      \item Exemple avec $m=43$
      \item "secret" $\leadsto$ 21
      \item "cretes" $\leadsto$ 21
      \item "cat" $\leadsto$ 21
      \item "av" $\leadsto$ 21
      \item "va" $\leadsto$ 21
      \end{itemize}
    \end{alertblock}
  }
  \only<3>{
    \begin{block}{Bonne fonction~: $h(w) = \sum_{i=0}^{p-1}a_i\times26^i \mod m$}
      \begin{itemize}
      \item Exemple avec $m=43$
      \item "secret" $\leadsto$ 11
      \item "cretes" $\leadsto$ 25
      \item "cat" $\leadsto$ 26
      \item "av" $\leadsto$ 22
      \item "va" $\leadsto$ 13
      \end{itemize}
    \end{block}
  }
  \only<4->{
    Comment calculer $h(w) = \sum_{i=0}^{p-1}a_i\times26^i \mod m$~?
    \begin{itemize}
    \item<4-> Problème~: éviter de calculer explicitement $26^i$
    \item<5-> En Java, calcul de $26^i$ en entier OK jusqu'à $i=6$
    \item<6-> Solution~: forme de Horner
    \item<7-> $h(w) = a_0 + 26(a_1 + 26(a_2 + 26 (\ldots )))$
    \item<7-> Calculé en Java avec une boucle
    \end{itemize}
  }
\end{frame}
